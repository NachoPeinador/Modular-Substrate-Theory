\subsection{The Double Observational Fracture}

Twenty-first century physics faces two seemingly unrelated anomalies that challenge established theoretical frameworks. In the cosmological domain, the \textbf{Hubble tension} has reached an unprecedented statistical significance level ($\sim 5\sigma$) between cosmic microwave background (CMB) inferences by the Planck collaboration \cite{Planck2018} and local measurements using Cepheids and Type Ia supernovae by SH0ES \cite{Riess2022}. This discrepancy, recently confirmed by James Webb Space Telescope observations that rule out systematic errors in Cepheid photometry \cite{Riess2024}, suggests a fundamental fracture in our understanding of cosmic expansion.

Simultaneously, in the particle physics domain, the \textbf{emergence of exotic hadrons} like the hexaquark $d^*(2380)$ \cite{Bashkanov2024} and the tetraquark $T_{cc}^+$ \cite{Harada2025} challenges conventional Standard Model taxonomy. These states possess properties (compact radii $\sim 0.7$ fm, hidden color composition) that don't naturally fit into the quarkonia or hadronic molecule schemes, pointing to new confinement mechanisms beyond the SU(3) color model of quantum chromodynamics (QCD).

To these tensions adds the \textbf{$S_8$ tension} in weak lensing measurements by KiDS-1000 \cite{KiDS2021} and DES \cite{DES2022}, indicating less matter clustering than predicted by $\Lambda$CDM. This triple anomaly ($H_0$, exotic hadrons, $S_8$) constitutes what we might call an \textbf{epistemological crisis} in fundamental physics: existing theories, successful in their original domains, show cracks when subjected to increasing precision tests.

Independent critical validation comes from galactic kinematics: recent analyses of the \textit{CosmicFlows-4} catalog \cite{Mazurenko2024} rule out the existence of "giant voids" of density (like the $\sim 300$ Mpc KBC model) by imposing an upper limit of $\sim 70-100$ Mpc for any local structure compatible with observed peculiar flows. Any solution to the Hubble tension must therefore operate within this strict scale, ruling out an entire class of phenomenological models.

\section{Computational Methodology}
\label{app:computational}

This appendix details the algorithms and numerical methods used for the Bayesian inference and data analysis presented in the article.

\subsection{Bayesian Inference Algorithm with Physical Priors}
\label{app:bayesian_inference_algorithm}

\begin{algorithm}[htbp]
\SetAlgoLined
\KwData{Observational data: $\mathcal{D} = \{H(z_i), \mu_{\text{SN}}(z_j), \text{BAO}_k\}$}
\KwResult{Posterior distribution $p(\theta|\mathcal{D})$ for $\theta = (z_c, \Delta z)$}
\SetKwFunction{FLogLike}{LogLikelihood}
\SetKwFunction{FLogPrior}{LogPrior}
\SetKwFunction{FLogPost}{LogPosterior}
\SetKwProg{Fn}{Function}{:}{}

\Fn{\FLogLike{$\theta$}}{
    \tcp{Calculate $H_{\text{MST}}(z)$ using eq. (20)}
    $H_{\text{pred}}(z) \gets H_{\Lambda\text{CDM}}(z) / \sqrt{1 - \Kinfo \Theta(z; \theta)}$\;
    
    \tcp{Likelihood for $H(z)$ data}
    $\mathcal{L}_H \gets \prod_i \mathcal{N}(H_{\text{obs}}(z_i)|H_{\text{pred}}(z_i), \sigma_{H,i})$\;
    
    \tcp{Likelihood for supernovae}
    $\mu_{\text{pred}}(z) \gets 5\log_{10}\left(\frac{c}{H_0}\int_0^z \frac{dz'}{H_{\text{pred}}(z')}\right) + 25$\;
    $\mathcal{L}_{\text{SN}} \gets \prod_j \mathcal{N}(\mu_{\text{obs}}(z_j)|\mu_{\text{pred}}(z_j), \sigma_{\mu,j})$\;
    
    \tcp{Likelihood for BAO}
    $\mathcal{L}_{\text{BAO}} \gets \prod_k \mathcal{N}(r_d^{\text{obs}}H_{\text{pred}}(z_k)|r_d^{\text{fid}}H_{\text{obs}}(z_k), \sigma_{\text{BAO},k})$\;
    
    \Return $\ln\mathcal{L}_H + \ln\mathcal{L}_{\text{SN}} + \ln\mathcal{L}_{\text{BAO}}$\;
}

\Fn{\FLogPrior{$\theta$}}{
    \tcp{Uniform priors in physical ranges}
    \If{$z_c \notin [0.005, 0.035]$ \textbf{or} $\Delta z \notin [0.001, 0.020]$}{
        \Return $-\infty$\;
    }
    
    \tcp{Kinematic prior from CosmicFlows-4}
    $D_c \gets \int_0^{z_c} \frac{c}{H_{\text{pred}}(z')} dz'$\;
    \If{$D_c > 70.0\ \si{\mega\parsec}$}{
        \tcp{Soft Gaussian penalty}
        $\ln\pi_{\text{CF4}} \gets -0.5\left(\frac{D_c - 70.0}{5.0}\right)^2$\;
    }\Else{
        $\ln\pi_{\text{CF4}} \gets 0$\;
    }
    
    \tcp{FSS smoothing prior}
    $\ln\pi_{\text{FSS}} \gets -0.5\left(\frac{\Delta z - 0.014}{0.002}\right)^2$\;
    
    \Return $\ln\pi_{\text{CF4}} + \ln\pi_{\text{FSS}}$\;
}

\Fn{\FLogPost{$\theta$}}{
    $\mathcal{L} \gets \FLogLike{$\theta$}$\;
    $\pi \gets \FLogPrior{$\theta$}$\;
    \If{$\pi = -\infty$}{
        \Return $-\infty$\;
    }
    \Return $\mathcal{L} + \pi$\;
}

\tcp{MCMC sampling with emcee}
Initialize 32 walkers in plausible region\;
\For{$i \gets 1$ \KwTo $10^4$}{
    Propose new $\theta'$ from proposal distribution\;
    $\alpha \gets \min\left(1, \frac{\FLogPost{$\theta'$}}{\FLogPost{$\theta$}}\right)$\;
    \If{$u \sim \mathcal{U}(0,1) < \alpha$}{
        Accept $\theta'$\;
    }
}
\caption{Bayesian inference for MST parameters}
\label{alg:bayesian_inference}
\end{algorithm}

\subsubsection{Bayesian Information Criterion Calculation}

The Bayesian model comparison in Section \ref{subsec:bayesian_analysis} uses the standard BIC formulation:

\begin{definition}[BIC for Model Comparison]
For a model with maximum log-likelihood $\ln\mathcal{L}_{\text{max}}$ and $k$ parameters fitting $n$ data points:
\begin{equation}
\text{BIC} = k \ln(n) - 2 \ln(\mathcal{L}_{\text{max}})
\end{equation}
The difference $\Delta\text{BIC} = \text{BIC}_{\text{model}} - \text{BIC}_{\Lambda\text{CDM}}$ quantifies relative evidence.
\end{definition}

\begin{proposition}[Data Points for Cosmological Comparison]
The cosmological comparison uses $n = 8$ independent data points:
\begin{align}
n_{\text{data}} &= n_{H(z)} + n_{H_0} + n_{S_8} \\
&= 6\ (H(z)\ \text{measurements}) + 1\ (H_0^{\text{SH0ES}}) + 1\ (S_8^{\text{KiDS}}) = 8
\end{align}
\end{proposition}

\begin{proposition}[MST Log-Likelihood Calculation]
The maximum log-likelihood for MST with optimal parameters $(z_c, \Delta z) = (0.01703, 0.01453)$ is:
\begin{equation}
\ln\mathcal{L}_{\text{max}}^{\text{MST}} = -12.06
\end{equation}
derived from the sum of Gaussian likelihoods for all data points.
\end{proposition}

\subsection{Hadronic Compression Analysis}
\label{app:hadronic_analysis}

\begin{algorithm}[H]
\SetAlgoLined
\KwData{Hadronic mass table from PDG 2024}
\KwResult{Compression factor $\Gamma$ and statistical significance}
\SetKwFunction{FAnalyze}{AnalyzeCompression}
\SetKwProg{Fn}{Function}{:}{}

\Fn{\FAnalyze{}}{
    \tcp{Load data for each family}
    \ForEach{family $\in$ $\{\Upsilon(1S,2S,3S), \psi(1S,2S,3S), \ldots\}$}{
        $M_1, M_2, M_3 \gets$ ground state masses\;
        $\sigma_1, \sigma_2, \sigma_3 \gets$ uncertainties\;
        
        \tcp{Calculate experimental ratio}
        $R_{\text{exp}} \gets \frac{M_3 - M_2}{M_2 - M_1}$\;
        $\sigma_R \gets R_{\text{exp}}\sqrt{\left(\frac{\sigma_3}{M_3-M_2}\right)^2 + \left(\frac{\sigma_2+\sigma_1}{M_2-M_1}\right)^2}$\;
        
        \tcp{Calculate theoretical Airy ratio}
        $z_1, z_2, z_3 \gets$ zeros of $Ai(-z)$ (2.33811, 4.08795, 5.52056)\;
        $R_{\text{Airy}} \gets \frac{z_3 - z_2}{z_2 - z_1} \approx 0.81887$\;
        
        \tcp{Compression factor}
        $\Gamma \gets \frac{R_{\text{exp}}}{R_{\text{Airy}}}$\;
        $\sigma_\Gamma \gets \frac{\sigma_R}{R_{\text{Airy}}}$\;
        
        \tcp{Match with $\beta = 0.75$}
        $\Gamma_{\text{theo}} \gets R_{\text{Airy}}^{\beta-1} \approx 0.81887^{-0.25} \approx 0.7425$\;
        $\text{match} \gets 100 \times \left(1 - \frac{|\Gamma - \Gamma_{\text{theo}}|}{\Gamma_{\text{teo}}}\right)$\;
        
        \tcp{Statistical test}
        $t_{\text{null}} \gets \frac{\Gamma - 1.0}{\sigma_\Gamma}$ \tcp{vs no compression}
        $t_{\text{MST}} \gets \frac{\Gamma - \Gamma_{\text{theo}}}{\sigma_\Gamma}$ \tcp{vs MST}
        
        Save results\;
    }
    
    \tcp{Combined analysis}
    $\Gamma_{\text{global}} \gets$ weighted average by $\sigma_\Gamma^{-2}$\;
    $\sigma_{\text{global}} \gets$ combined standard error\;
    
    \tcp{Confidence interval}
    $\Gamma_{\text{CI}} \gets [\Gamma_{\text{global}} - 1.96\sigma_{\text{global}}, \Gamma_{\text{global}} + 1.96\sigma_{\text{global}}]$\;
    
    \Return $\Gamma_{\text{global}}, \sigma_{\text{global}}, \Gamma_{\text{CI}}$\;
}
\caption{Dimensional compression analysis in hadronic spectra}
\label{alg:hadronic_compression}
\end{algorithm}
